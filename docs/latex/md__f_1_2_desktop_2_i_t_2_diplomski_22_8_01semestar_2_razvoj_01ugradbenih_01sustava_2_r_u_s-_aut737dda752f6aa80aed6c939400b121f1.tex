Hvala što želite doprinijeti ovom projektu! Prije nego što započnete, molimo vas da pročitate sljedeće smjernice kako bismo osigurali organiziran i učinkovit razvoj projekta.\hypertarget{md__f_1_2_desktop_2_i_t_2_diplomski_22_8_01semestar_2_razvoj_01ugradbenih_01sustava_2_r_u_s-_aut737dda752f6aa80aed6c939400b121f1_autotoc_md8}{}\doxysection{\texorpdfstring{Pravila doprinosa}{Pravila doprinosa}}\label{md__f_1_2_desktop_2_i_t_2_diplomski_22_8_01semestar_2_razvoj_01ugradbenih_01sustava_2_r_u_s-_aut737dda752f6aa80aed6c939400b121f1_autotoc_md8}

\begin{DoxyItemize}
\item Prije bilo kakvih promjena otvorite {\bfseries{Issue}} kako biste predložili izmjene ili poboljšanja.
\item Forkajte repozitorij i kreirajte vlastitu granu za svaku novu značajku ili ispravak greške.
\item Pišite čitljiv i uredan kod, u skladu sa standardima jezika (C++/\+Arduino stil kodiranja).
\item Prije otvaranja Pull Requesta, testirajte svoje izmjene kako biste izbjegli neželjene greške.
\item U opis Pull Requesta jasno napišite\+:
\begin{DoxyItemize}
\item Što ste promijenili.
\item Zašto je promjena potrebna.
\item Eventualne povezane probleme (\#issue-\/broj).
\end{DoxyItemize}
\end{DoxyItemize}\hypertarget{md__f_1_2_desktop_2_i_t_2_diplomski_22_8_01semestar_2_razvoj_01ugradbenih_01sustava_2_r_u_s-_aut737dda752f6aa80aed6c939400b121f1_autotoc_md9}{}\doxysection{\texorpdfstring{Stil komunikacije}{Stil komunikacije}}\label{md__f_1_2_desktop_2_i_t_2_diplomski_22_8_01semestar_2_razvoj_01ugradbenih_01sustava_2_r_u_s-_aut737dda752f6aa80aed6c939400b121f1_autotoc_md9}

\begin{DoxyItemize}
\item Budite ljubazni i profesionalni u komunikaciji.
\item Kritike neka budu konstruktivne, s prijedlogom poboljšanja.
\end{DoxyItemize}\hypertarget{md__f_1_2_desktop_2_i_t_2_diplomski_22_8_01semestar_2_razvoj_01ugradbenih_01sustava_2_r_u_s-_aut737dda752f6aa80aed6c939400b121f1_autotoc_md10}{}\doxysection{\texorpdfstring{Kako početi?}{Kako početi?}}\label{md__f_1_2_desktop_2_i_t_2_diplomski_22_8_01semestar_2_razvoj_01ugradbenih_01sustava_2_r_u_s-_aut737dda752f6aa80aed6c939400b121f1_autotoc_md10}

\begin{DoxyEnumerate}
\item Forkajte projekt
\item Klonirajte svoj fork\+: git clone \href{https://github.com/va\%C5\%A1e-korisni\%C4\%8Dko-ime/ime-projekta.git}{\texttt{ https\+://github.\+com/vaše-\/korisničko-\/ime/ime-\/projekta.\+git}}
\item Napravite novu granu\+: git checkout -\/b naziv-\/grane
\item Unesite promjene i počinite\+: git commit -\/m "{}\+Opis promjene"{}
\item Pushajte granu\+: git push origin naziv-\/grane
\item Otvorite Pull Request na originalnom repozitoriju.
\end{DoxyEnumerate}
